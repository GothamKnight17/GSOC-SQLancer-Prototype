for(int i = 1; i < n; i++){ 
    key = arr[i];
    int j = i - 1;
    while(j>=0 && arr[j]>key){
        arr[j+1]=arr[j];
        j--;
    }
    arr[j+1] = key;
}


for(int i=0; i<n-1; i++){ 
    int curr = i;
    for(int j=i+1; j<n; j++){
        if(arr[curr] > arr[j]){
            curr = i;
        }
    }
}
swap(arr, curr, i)
